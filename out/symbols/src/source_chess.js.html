1 /**
  2  * Creates a chess board with '*' and ' ' symbols.
  3  * @function
  4  * @param {number} dim - Dimension of the chess board.
  5  * @returns {string} A string representing a chess board
  6  * with '*', ' ' and '\n' symbols.
  7 */
  8 const chess = dim => {
  9     if (!Number.isInteger(Number(dim))) return null;
 10     if (dim < 2) return null;
 11     const pattern1 = '* ';
 12     const pattern2 = ' *';
 13     let row1 = pattern1.repeat(Math.floor(dim/2));
 14     let row2 = pattern2.repeat(Math.floor(dim/2));
 15     const normalizeRow = row => (dim%2 !== 0 ? row + row[0] : row) + '\n';
 16     row1 = normalizeRow(row1);
 17     row2 = normalizeRow(row2);
 18     let board = (row1 + row2).repeat(Math.floor(dim/2));
 19     return dim%2 !== 0 ? board + row1 : board;
 20 }
